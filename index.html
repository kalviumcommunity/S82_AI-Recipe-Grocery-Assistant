import dotenv from "dotenv"; dotenv.config(); import { GoogleGenerativeAI } from "@google/generative-ai"; const genAI = new GoogleGenerativeAI(process.env.GEMINI_API_KEY); async function run() { const model = genAI.getGenerativeModel({ model: "gemini-1.5-flash" }); const result = await model.generateContent("Hello Gemini!"); console.log(result.response.text()); } run();
